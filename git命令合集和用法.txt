git init//只执行一次，创建.git文件夹。管理项目。git文件夹是隐藏的

git status//可以执行无数次，查看文件状态，可以随便查看

git clone .git filename  //克隆文件，将当前的分支中的文件复制到Filename这么文件夹下

git add .  // 把新加文件加到GIT，让GIT管理

git status  //查看文件状态

git commit -m "说明信息" //添加更新信息

git remote add origin http://manage.hand-south.com/diffusion/GITHUANGZILIANG/git_huangziliang.git // 添加远程服务器地址，origin是远程服务器名称。

git remote -v  //查看该项目的远程服务器

git push origin name:name   //将分支文件推送到服务器上的分支。 其中origin 是路径的标识，可以用git remote -v查看，第一个name是指本地的分支名，第二个是指网上的分支名

git clone https://github.com/hailor/TestGit.git projectA  //projectA就是项目放的地方，系统会自动创建。

git pull  //用于更新文件，从服务器中下载文件进行更新。

git branch -a  //查看分支，显示的绿色为本地分支，红色为服务器上的分支。

git branch name //创建本地分支，name为分支名称

git checkout name  //切换分支

git merge name     //将分支合并到本地分支，也可以将分支名中的分支合并到当前的分支

git checkout -b name origin/name  //下载分支并且切换到该分支
